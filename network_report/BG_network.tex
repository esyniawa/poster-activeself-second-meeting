\end{dmath*}

\begin{dmath*}
{{\text{condition}}_0}(t) = \begin{cases}1\qquad \text{if} \quad {w}(t) > 0.0 \wedge {{\text{trace}}}(t) > 0.0\\ 0 \qquad \text{otherwise.} \end{cases}
\end{dmath*}

\begin{dmath*}
{{\text{dopa}}_{\text{mod}}}(t) = \begin{cases}K_{\text{burst}} \cdot {\text{DA}}_{\text{type}} \cdot {{\text{dopa}}_{\text{sum}}}(t)\qquad \text{if} \quad {\text{DA}}_{\text{type}} \cdot {{\text{dopa}}_{\text{sum}}}(t) > 0\\ K_{\text{dip}} \cdot {\text{DA}}_{\text{type}} \cdot {{\text{condition}}_0}(t) \cdot {{\text{dopa}}_{\text{sum}}}(t) \qquad \text{otherwise.} \end{cases}
\end{dmath*}

\begin{dmath*}
{\delta}(t) = - {\alpha}(t) \cdot \left(- {\text{threshold}}_{\text{post}} + {{r}^{\text{post}}}(t) - \operatorname{mean}{\left({{r}^{\text{post}}}(t) \right)}\right)^+ + {{\text{dopa}}_{\text{mod}}}(t) \cdot {{\text{trace}}}(t)
\end{dmath*}
